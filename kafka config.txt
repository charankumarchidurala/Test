public class KafkaConsumerConfig extends CommonKafkaConsumerConfig {

	@Value(("${kafka.item.group}"))
	String itemGroup;

	@Bean("itemKafkaListenerContainerFactory")
	@RefreshScope
	public KafkaListenerContainerFactory<ConcurrentMessageListenerContainer<String, String>> itemKafkaListenerContainerFactory() {
		ConcurrentKafkaListenerContainerFactory<String, String> factory = genericFactory();
		factory.setConsumerFactory(new DefaultKafkaConsumerFactory<>(itemConsumerConfigs()));
		return factory;
	}

	private Map<String, Object> itemConsumerConfigs() {
		Map<String, Object> propsMap = genericConsumerConfigs();
		propsMap.put(ConsumerConfig.GROUP_ID_CONFIG, itemGroup);
		return propsMap;
	}
}
